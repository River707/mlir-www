9 // Perform inplace bufferization within function boundaries.
10 // This is a specialized pass that supports inplace analysis for a fixed subset
11 // of ops that have well-defined inplace semantics.
12 // This pass caters to high-performance codegen where buffer reuse is deemed
13 // necessary: the pass should fail if the bufferized form of the function needs
14 // to return any buffer.
15 // Generic control-flow and branching are unsupported.
16 // Composability with extensible set of ops is not a first-class concern.
17 //
18 // Bufferization occurs by:
19 // a. performing an inPlace analysis `inPlaceAnalysisFuncOpInternals`
20 // which marks each operation within the function with the
21 // `kInPlaceResultsAttrName` attribute.
22 // b. traversing each operation in the function and rewriting it in
23 // buffer form and keeping a BlockAndValueMapping mapping of the
24 // rewrites. New allocations are introduced during this step.
25 // TODO: Allocation + depending op hoisting to outermost enclosing
26 // sequential scope.
27 // c. at the end of this bufferization, 2 cases may occur:
28 // * inplaceable function arguments may be reused in place after the
29 // function itself has been bufferized. This is encoded by IR resembling:
30 //
31 // ```
32 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
33 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true}) -> tensor<?xf32> {
34 // %0 = memref.buffer_cast %A : memref<?xf32, #map>
35 // // ... uses of %0
36 // %res = memref.tensor_load %0 : memref<?xf32, #map>
37 // return %res : tensor<?xf32>
38 // }
39 // ```
40 //
41 // this is the cue for the bufferization of the function foo (and calls to
42 // it) may bufferize to `func @foo(%A: memref<?xf32, some_layout>)`.
43 // To fully achieve bufferization, an additional analysis is needed to
44 // determine whether function argument/operand pairs bufferize to a single
45 // inplace buffer argument (i.e. functions may return tensors in arbitrary
46 // order that may not match argument numbers).
47 // * results that don't map to an inplaceable function argument must be
48 // allocated. Since memref semantics wrt ownership of the underlying
49 // memory region are not well-defined, comprehensive bufferization chooses
50 // to perform allocations in a scoped fashion: returning memrefs is always
51 // considered illegal. Such scenarios are encoded by IR resembling:
52 //
53 // ```
54 // #map = affine_map<(d0)[s0, s1] -> (d0 * s1 + s0)>
55 // func @foo(%A: tensor<?xf32> {linalg.inplaceable = true}) -> tensor<?xf32> {
56 // %0 = memref.buffer_cast %A : memref<?xf32, #map>
57 // %1 = memref.dim %0, %c0 : memref<?xf32, #map>
58 // %2 = memref.alloc(%1) : memref<?xf32>
59 // %3 = memref.cast %2 : memref<?xf32> to memref<?xf32, #map>
60 // // ... uses of %3
61 // memref.dealloc %2 : memref<?xf32, #map>
62 // %res = memref.tensor_load %3 : memref<?xf32, #map>
63 // return %res : tensor<?xf32>
64 // }
65 // ```
66 //
67 // this is the cue for the bufferization of the function foo (and calls to
68 // it) that it must bufferize to
69 // `func @foo(%A: memref<?xf32, some_layout>,
70 // %B: memref<?xf32, some_layout>)` (i.e. make a cloned
71 // allocation of the result tensor)
72 // To fully achieve bufferization, the alloc/dealloc pair must be lifted
73 // out of the function at each call site.
74 //
75 // Lastly, note that layout map chosen to bufferize is the most dynamic
76 // canonical strided layout of the proper rank. This ensures compatibility with
77 // expected layouts after transformations. Combinations of memref.cast +
78 // canonicalization are responsible for clean ups.

106 /// Return the OpResult that matches an operand.
107 /// Return null if no such result exists.
108 OpResult getMatchingOpResult(LinalgOp linalgOp, OpOperand &opOperand) {
109  if (!opOperand.get().getType().isa<RankedTensorType>())
110  return OpResult();
111  // For now assume inputs are never inplaceable.
112  // TODO: refine this.
113  if (opOperand.getOperandNumber() < linalgOp.getNumInputs())
114  return OpResult();
115  // For now assume if the operand appears twice, it is not inplaceable.
116  // TODO: refine this.
117  for (auto &opOperand2 : linalgOp->getOpOperands()) {
118  if (opOperand.getOperandNumber() == opOperand2.getOperandNumber())
119  continue;
120  if (opOperand.get() == opOperand2.get())
121  return OpResult();
122  }
123  int64_t outputOperandIndex =
124  opOperand.getOperandNumber() - linalgOp.getNumInputs();
125  int64_t numOutputBuffers = 0;
126  for (unsigned idx = 0; idx < outputOperandIndex; ++idx)
127  if (!linalgOp.getOutputOperand(idx)->get().getType().isa<TensorType>())
128  ++numOutputBuffers;
129  return linalgOp->getResult(outputOperandIndex - numOutputBuffers);
130 }

150 /// Determine which results may be reused inplace by the bufferization
151 /// patterns of `bufferizeFuncOpInternals`.
152 /// The inplace analysis uses this information along with interfering read
153 /// analysis to determine which op results reuse the same buffer as some
154 /// operand.
155 OpResult getMatchingOpResult(OpOperand &opOperand) {
156  return llvm::TypeSwitch<Operation *, OpResult>(opOperand.getOwner())
157  // clang-format off
158  // Ops that perform destructive updates on operand(s) to produce
159  // result(s).

161  SubTensorInsertOp,
162  VectorTransferOpInterface>(
163  [&](auto op) { return getMatchingOpResult(op, opOperand); })
164  // Other ops.
165  .Case<SubTensorOp>([&](auto op) { return OpResult(); })
166  .Default([&](Operation *op) { return OpResult(); });
167  // clang-format on

204 /// Mark whether OpResult can actually be bufferized inplace. If `inPlace` is
205 /// `InPlaceSpec::True`, the use-def chain analysis has guaranteed that no
206 /// subsequent write would occur to the bufferized tensor value (i.e. the result
207 /// can be bufferized inPlace).
208 static void setInPlaceOpResult(OpResult opResult,
209  InPlaceSpec inPlace = InPlaceSpec::True) {
210  if (!opResult)
211  return;
212 
213  Operation *op = opResult.getOwner();
214  auto attr =
215  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
216  SmallVector<StringRef> inPlaceVector =
217  attr ? SmallVector<StringRef>(
218  llvm::to_vector<4>(attr.getAsValueRange<StringAttr>()))
219  : SmallVector<StringRef>(op->getNumResults(),
220  stringify(InPlaceSpec::None));
221  LLVM_DEBUG(DBGS() << "Set inPlace=" << stringify(inPlace) << ": " << *op
222  << " @idx=" << opResult.getResultNumber() << "\n");
223  inPlaceVector[opResult.getResultNumber()] = stringify(inPlace);
224  op->setAttr(kInPlaceResultsAttrName,
225  OpBuilder(op).getStrArrayAttr(inPlaceVector));
226 }

228 /// Get the InPlaceSpec attribute entry `kInPlaceResultsAttrName` for
229 /// `opResult`. If the result is `InPlaceSpec::True`, the use-def chain analysis
230 /// has guaranteed that no subsequent read of the tensor value occurs and the
231 /// result can be buferized inPlace.
232 /// If no InPlaceSpec attribute has been set for `opResult`, return
233 /// InPlaceSpec::None.
234 static InPlaceSpec getInPlace(OpResult opResult) {
235  if (!opResult)
236  return InPlaceSpec::None;
237 
238  Operation *op = opResult.getOwner();
239  auto attr =
240  op->getAttr(kInPlaceResultsAttrName).dyn_cast_or_null<ArrayAttr>();
241  if (!attr)
242  return InPlaceSpec::None;
243 
244  // Must return a proper value.
245  return *symbolize(*(attr.getAsValueRange<StringAttr>().begin() +
246  opResult.getResultNumber()));
247 }

250 /// If it does not come from a function, return InPlaceSpec::False.
251 static InPlaceSpec getInPlace(BlockArgument bbArg) {

330 /// Return a MemRefType to which the `tensorType` can be bufferized in a
331 /// composable fashion. The layout must be the most dynamic possible and
332 /// canonicalize away once bufferization is finished.
333 static MemRefType getDynamicMemRefType(RankedTensorType tensorType,
334  unsigned addressSpace = 0) {
335  // TODO: address space decisions to connect with the actual alloc.
336  int64_t dynamicOffset = ShapedType::kDynamicStrideOrOffset;
337  SmallVector<int64_t> dynamicStrides(tensorType.getRank(),
338  ShapedType::kDynamicStrideOrOffset);
339  AffineMap stridedLayout = makeStridedLinearLayoutMap(
340  dynamicStrides, dynamicOffset, tensorType.getContext());
341  return MemRefType::get(tensorType.getShape(), tensorType.getElementType(),
342  stridedLayout, addressSpace);
343 }

349 /// Create an Allocop/DeAllocOp pair, where the AllocOp is after
350 /// `shapedValue.getDefiningOp` (or at the top of the block in case of a bbArg)
351 /// and the DeallocOp is at the end of the block.
352 static Value createNewAllocDeallocPairForShapedValue(
353  OpBuilder &b, Location loc, Value shapedValue,
354  SmallVector<Value, 4> dynOperands = {}) {
355  // Take a guard before anything else.
356  OpBuilder::InsertionGuard g(b);
357 
358  // TODO: non-zero address space.
359  // TODO: layout information if relevant.
360  // Cannot allocate an unranked memref so just always go for the contiguous
361  // form.
362  MemRefType allocMemRefType =
363  getContiguousMemRefType(shapedValue.getType().cast<ShapedType>());
364  assert(shapedValue.getType().isa<ShapedType>());
365  MemRefType memRefType = shapedValue.getType().dyn_cast<MemRefType>();
366  memRefType = memRefType ? memRefType : allocMemRefType;
367 
368  if (auto bbArg = shapedValue.dyn_cast<BlockArgument>()) {
369  b.setInsertionPointToStart(bbArg.getOwner());
370  loc = bbArg.getOwner()->getParentOp()->getLoc();
371  } else {
372  b.setInsertionPointAfter(shapedValue.getDefiningOp());
373  loc = shapedValue.getDefiningOp()->getLoc();
374  }
375 
376  // If the dynOperands are not passed explicity, copmpute them.
377  // This circumvents currently missing dim(init_tensor) canonicalizations.
378  // TODO: dim(init_tensor) canonicalization.
379  if (dynOperands.empty()) {
380  for (auto dim : llvm::enumerate(memRefType.getShape()))
381  if (dim.value() == ShapedType::kDynamicSize)
382  dynOperands.push_back(
383  b.create<memref::DimOp>(loc, shapedValue, dim.index()));
384  }
385 
386  Value allocated =
387  b.create<memref::AllocOp>(loc, allocMemRefType, dynOperands);
388  Value casted = allocated;
389  if (memRefType != allocMemRefType)
390  casted = b.create<memref::CastOp>(loc, memRefType, allocated);
391  b.setInsertionPoint(allocated.getParentBlock()->getTerminator());
392  b.create<memref::DeallocOp>(loc, allocated);
393  return casted;
394 }

400 /// Helper function for LinalgOp bufferization.
401 /// Operate on mixed tensor + buffer Linalg ops for progressive bufferization.
402 /// Allocate the output buffers for the remaining tensor output operands of
403 /// the Linalg op. If the tensor is an "init" tensor (i.e. its value is
404 /// actually used in the payload region), we additionally copy the original
405 /// value into the newly allocated buffer.
406 static LogicalResult
407 allocateBuffersForResults(OpBuilder &b, Location loc, LinalgOp op,
408  SmallVectorImpl<Value> &resultBuffers,
409  BlockAndValueMapping &bvm) {
410  // Take a guard before anything else.
411  OpBuilder::InsertionGuard g(b);
412 
413  // Lazily compute loopRanges.
414  SmallVector<Range, 4> loopRanges;
415 
416  // Linalg invariant: output tensors and result match 1-1.
417  assert(op.getOutputTensorOperands().size() == op->getNumResults());
418  for (OpOperand *opOperand : op.getOutputOperands()) {
419  Value output = opOperand->get();
420  if (output.getType().isa<MemRefType>()) {
421  resultBuffers.push_back(output);
422  continue;
423  }
424 
425  // If output tensor is marked inPlace, just use the buffer.
426  // The following uses internal knowledge of the position of tied operand /
427  // results.
428  OpResult tiedResult = getMatchingOpResult(op, *opOperand);
429  if (getInPlace(tiedResult) == InPlaceSpec::True) {
430  Value v = lookup(bvm, output);
431  if (!v)
432  return failure();
433  resultBuffers.push_back(v);
434  continue;
435  }
436 
437  Value dimTensor = bvm.lookupOrDefault(output);
438  Value alloc = createNewAllocDeallocPairForShapedValue(b, loc, dimTensor);
439  b.setInsertionPointAfter(alloc.getDefiningOp());
440  resultBuffers.push_back(alloc);
441 
442  // Additionally, if the output buffer is used, clone its value for now.
443  if (op.payloadUsesValueFromOperand(opOperand)) {
444  Value v = lookup(bvm, output);
445  if (!v)
446  return failure();
447  b.create<CopyOp>(loc, v, alloc);
448  }
449  }
450  if (op->getNumResults())
451  map(bvm, op->getResults(), resultBuffers);
452 
453  return success();
454 }

456 static void finalizeBufferAllocation(OpBuilder &b, LinalgOp op,
457  ValueRange inputs, ValueRange outputs,
458  BlockAndValueMapping &bvm) {
459  SmallVector<Value, 8> newOperands = inputs;
460  newOperands.append(outputs.begin(), outputs.end());
461  auto otherOperands = op.getAssumedNonShapedOperands();
462  newOperands.append(otherOperands.begin(), otherOperands.end());
463  Location loc = op.getLoc();
464  op.clone(b, loc, /*resultTypes=*/TypeRange{}, newOperands);
465 
466  // Replace the results of the old op with the new output buffers.
467  if (op->getNumResults())
468  map(bvm, op->getResults(), outputs);
469  if (!op.hasTensorSemantics())
470  op->erase();
471 }

473 /// Generic conversion for any LinalgOp.
474 /// Operate on mixed tensor + buffer Linalg ops for progressive bufferization.
475 static LogicalResult bufferize(OpBuilder &b, LinalgOp op,
476  BlockAndValueMapping &bvm) {
477  // Take a guard before anything else.
478  OpBuilder::InsertionGuard g(b);
479 
480  if (op.hasBufferSemantics())
481  return failure();
482 
483  LLVM_DEBUG(DBGS() << "bufferize: " << *op << "\n");
484 
485  b.setInsertionPoint(op);
486  Location loc = op.getLoc();
487  SmallVector<Value, 2> newInputBuffers;
488  newInputBuffers.reserve(op.getNumInputs());
489  for (OpOperand *opOperand : op.getInputOperands()) {
490  Value v = lookup(bvm, opOperand->get());
491  if (!v)
492  return failure();
493  newInputBuffers.push_back(v);
494  }
495  SmallVector<Value, 2> newOutputBuffers;
496  if (failed(allocateBuffersForResults(b, loc, op, newOutputBuffers, bvm)))
497  return failure();
498  finalizeBufferAllocation(b, op, newInputBuffers, newOutputBuffers, bvm);
499  return success();
500 }

515 /// FuncOp always creates TensorToMemRef ops.
516 static LogicalResult bufferize(OpBuilder &b, FuncOp funcOp,
517  BlockAndValueMapping &bvm) {
518  // Take a guard before anything else.
519  OpBuilder::InsertionGuard g(b);
520  b.setInsertionPointToStart(&funcOp.body().front());
521  for (auto bbArg : funcOp.getArguments()) {
522  auto tensorType = bbArg.getType().dyn_cast<TensorType>();
523  if (!tensorType)
524  continue;
525  auto rankedTensorType = tensorType.dyn_cast<RankedTensorType>();
526  // Cast the tensor to the most dynamic buffer possible. Further
527  // canonicalizations will clean up.
528  Type memRefType = rankedTensorType
529  ? getDynamicMemRefType(rankedTensorType)
530  : getContiguousOrUnrankedMemRefType(tensorType);
531  Value tensorToMemref =
532  b.create<memref::BufferCastOp>(funcOp.getLoc(), memRefType, bbArg);
533  map(bvm, bbArg, tensorToMemref);
534  }
535  return success();
536 }

559 /// Bufferize SubTensorOp to subview with optional alloc + copy depending on
560 /// whether or not it is marked inplaceable.
561 /// Note that `getMatchingOpResult` on a SubTensorOp always returns null.
562 /// As consequence a SubTensorOp always alloc + copy when taken in isolation.
563 static LogicalResult bufferize(OpBuilder &b, SubTensorOp subTensorOp,
564  BlockAndValueMapping &bvm) {
565  LLVM_DEBUG(DBGS() << "bufferize: " << *subTensorOp << "\n");
566 
567  // Take a guard before anything else.
568  OpBuilder::InsertionGuard g(b);
569  b.setInsertionPoint(subTensorOp);
570 
571  Location loc = subTensorOp.getLoc();
572  // Bail if source was not bufferized.
573  Value srcMemref = lookup(bvm, subTensorOp.source());
574  if (!srcMemref)
575  return failure();
576  auto srcMemrefType = srcMemref.getType().cast<MemRefType>();
577  auto dstTensorType = subTensorOp.result().getType().cast<RankedTensorType>();
578 
579  // If not inplaceable, alloc.
580  Value alloc;
581  auto inPlace = getInPlace(subTensorOp->getResult(0));
582  if (inPlace != InPlaceSpec::True) {
583  alloc =
584  createNewAllocDeallocPairForShapedValue(b, loc, subTensorOp.result());
585  b.setInsertionPointAfter(alloc.getDefiningOp());
586  }
587 
588  // Bufferize to subview.
589  auto subviewMemRefType =
590  memref::SubViewOp::inferRankReducedResultType(
591  dstTensorType.getRank(), srcMemrefType, subTensorOp.getMixedOffsets(),
592  subTensorOp.getMixedSizes(), subTensorOp.getMixedStrides())
593  .cast<MemRefType>();
594  Value subView = b.create<memref::SubViewOp>(
595  loc, subviewMemRefType, srcMemref, subTensorOp.getMixedOffsets(),
596  subTensorOp.getMixedSizes(), subTensorOp.getMixedStrides());
597 
598  /// If not inplaceable, copy.
599  if (alloc) {
600  b.create<CopyOp>(subTensorOp.getLoc(), subView, alloc);
601  subView = alloc;
602  }
603 
604  map(bvm, subTensorOp.result(), subView);
605  return success();

608 static LogicalResult bufferize(OpBuilder &b,
609  SubTensorInsertOp subTensorInsertOp,
610  BlockAndValueMapping &bvm) {
611  LLVM_DEBUG(DBGS() << "bufferize: " << *subTensorInsertOp << "\n");
612 
613  // Take a guard before anything else.
614  OpBuilder::InsertionGuard g(b);
615  b.setInsertionPoint(subTensorInsertOp);
616  Location loc = subTensorInsertOp.getLoc();
617 
618  Value dstMemref = lookup(bvm, subTensorInsertOp.dest());
619  if (!dstMemref)
620  return failure();
621  auto inPlace = getInPlace(subTensorInsertOp->getResult(0));
622  if (inPlace != InPlaceSpec::True) {
623  // Since subtensor_insert arise from tiling and introducing loops, this case
624  // is generally a deal breaker. When used with loops, this ends up cloning
625  // the whole tensor on every single iteration and is a symptom of a
626  // catastrophically bad scheduling decision.
627  // TODO: be very loud about it or even consider failing the pass.
628  Value newDstMemref = createNewAllocDeallocPairForShapedValue(
629  b, loc, subTensorInsertOp.result());
630  b.setInsertionPointAfter(newDstMemref.getDefiningOp());
631  b.create<CopyOp>(subTensorInsertOp.getLoc(), dstMemref, newDstMemref);
632  dstMemref = newDstMemref;
633  }
634  auto dstMemrefType = dstMemref.getType().cast<MemRefType>();
635 
636  Value srcMemref = lookup(bvm, subTensorInsertOp.source());
637  if (!srcMemref)
638  return failure();
639  auto subviewMemRefType =
640  memref::SubViewOp::inferRankReducedResultType(
641  subTensorInsertOp.getSourceType().getRank(), dstMemrefType,
642  subTensorInsertOp.getMixedOffsets(),
643  subTensorInsertOp.getMixedSizes(),
644  subTensorInsertOp.getMixedStrides())
645  .cast<MemRefType>();
646 
647  // A copy of the source buffer is needed if either:
648  // - The producer of `source` is not inplace. This is the case where a
649  // subtensor is computed out of place into the inplace full tensor.
650  // - The result is not inplace. This is the case where the whole tensor is
651  // cloned and the clone needs to be updated.
652  Value source = subTensorInsertOp.source();
653  InPlaceSpec inPlaceProducer = InPlaceSpec::None;
654  if (auto opResult = source.dyn_cast<OpResult>())
655  inPlaceProducer = getInPlace(opResult);
656  else
657  inPlaceProducer = getInPlace(source.cast<BlockArgument>());
658  if (inPlaceProducer != InPlaceSpec::True) {
659  LLVM_DEBUG(DBGS() << "subtensor_insert needs extra source copy: " << source
660  << " -> copy\n");
661  // Take a subview of the dst.
662  Value subView = b.create<memref::SubViewOp>(
663  loc, subviewMemRefType, dstMemref, subTensorInsertOp.getMixedOffsets(),
664  subTensorInsertOp.getMixedSizes(), subTensorInsertOp.getMixedStrides());
665  b.create<CopyOp>(subTensorInsertOp.getLoc(), srcMemref, subView);
666  }
667 
668  map(bvm, subTensorInsertOp.result(), dstMemref);
669 
670  return success();
671 }

673 static LogicalResult bufferize(OpBuilder &b, VectorTransferOpInterface op,
674  BlockAndValueMapping &bvm) {
675  // Take a guard before anything else.
676  OpBuilder::InsertionGuard g(b);
677  b.setInsertionPoint(op);
678  Location loc = op.getLoc();
679 
680  if (op.getShapedType().isa<MemRefType>())
681  return failure();
682 
683  LLVM_DEBUG(DBGS() << "bufferize: " << *op << "\n");
684 
685  /// transfer_read from buffer always reads from the bufferized op.source().
686  if (auto readOp = dyn_cast<vector::TransferReadOp>(op.getOperation())) {
687  Value v = lookup(bvm, op.source());
688  if (!v)
689  return failure();
690  readOp.sourceMutable().assign(v);
691  return success();
692  }
693 
694  auto inPlace = getInPlace(op->getResult(0));
695  auto writeOp = cast<vector::TransferWriteOp>(op.getOperation());
696 
697  // If transfer_write is not inPlace, allocate a new buffer.
698  Value newInputBuffer;
699  if (inPlace != InPlaceSpec::True) {
700  newInputBuffer =
701  createNewAllocDeallocPairForShapedValue(b, loc, writeOp.result());
702  b.setInsertionPointAfter(newInputBuffer.getDefiningOp());
703  map(bvm, writeOp.result(), newInputBuffer);
704  } else {
705  // InPlace write will result in memref.tensor_load(x) which must
706  // canonicalize away with one of it uses.
707  newInputBuffer = lookup(bvm, writeOp.source());
708  if (!newInputBuffer)
709  return failure();
710  }
711 
712  // Create a new transfer_write on buffer that doesn't have a return value.
713  // Leave the previous transfer_write to dead code as it still has uses at
714  // this point.
715  b.create<vector::TransferWriteOp>(
716  loc, writeOp.vector(), newInputBuffer, writeOp.indices(),
717  writeOp.permutation_map(),
718  writeOp.in_bounds() ? *writeOp.in_bounds() : ArrayAttr());
719 
720  map(bvm, op->getResult(0), newInputBuffer);
721 
722  return success();
723 }

729 /// Determine whether any subsequent read of the tensor `opOperand` may occur.
730 /// For now, this assumes any use is a read. If any use of the tensor does not
731 /// properly dominate `opOperand.getOwner()`, then the tensor cannot be
732 /// bufferized inPlace.
733 // TODO: For now, this assumes any use is a read. Refine this.
734 bool hasInterferingTensorRead(OpOperand &opOperand,
735  const DominanceInfo &domInfo) {
736  if (!opOperand.get().getType().isa<RankedTensorType>())
737  return false;
738  for (auto &use : opOperand.get().getUses()) {
739  Operation *user = use.getOwner();
740  // If properly dominate, there is a clear sequence point and we can dismiss
741  // read.
742  if (domInfo.properlyDominates(user, opOperand.getOwner()))
743  continue;
744  // Otherwise, we need to analyze self-dependencies, for now just let it go.
745  // TODO: proper self-dependence analysis.
746  if (domInfo.dominates(user, opOperand.getOwner()))
747  continue;
748  if (user == opOperand.getOwner() &&
749  use.getOperandNumber() == opOperand.getOperandNumber())
750  continue;
751  LLVM_DEBUG(DBGS() << "found interfering read operand #"
752  << opOperand.getOperandNumber()
753  << " in op: " << *opOperand.getOwner() << "\n");
754  return true;
755  }
756  LLVM_DEBUG(DBGS() << "no interfering read\n");
757  return false;
758 }

760 /// Return false if either:
761 /// 1. `opOperand` is produced by a constant op. For now this is assumed to be
762 /// bufferized to a GlobalMemrefOp that cannot be written. Generalize in the
763 /// future.
764 /// 2.`opOperand` is a BlockArgument of a FuncOp that is not known to be
765 /// bufferizable inplace.
766 /// Return true otherwise.
767 static bool bufferizeToWriteable(OpOperand &opOperand) {
768  // Constant tensors are deemed not bufferizable for now.
769  if (auto constantOp =
770  dyn_cast_or_null<ConstantOp>(opOperand.get().getDefiningOp()))
771  return !constantOp.getResult().getType().isa<RankedTensorType>();
772  if (auto bbArg = opOperand.get().dyn_cast<BlockArgument>()) {
773  // Uses of function arguments that may not be written-to need to be copied.
774  // If the function argument itself is not inplaceable, early return false.
775  // If is is inplaceable, interfering tensor read need to be checked.
776  //
777  // TODO: better propagate the fact that we want a single clone inside the
778  // function. Atm every user that wants to write inplace will create its own
779  // alloc, irrespective of whether or not interfering reads occur.
780  if (isa<FuncOp>(bbArg.getOwner()->getParentOp())) {
781  if (getInPlace(bbArg) != InPlaceSpec::True)
782  return false;
783  } else {
784  // Conservatively dump any other block argument for now.
785  return false;
786  }
787  }
788  return true;
789 }

863 /// Analyze the internals of a FuncOp to determine inplaceable ops.
864 static void inPlaceAnalysisFuncOpInternals(FuncOp funcOp,
865  const DominanceInfo &domInfo) {
866  assert(funcOp && funcOp->getNumRegions() > 0 && !funcOp.body().empty() &&
867  "expected a funcOp definition with a body");

879  << opOperand.getOperandNumber() << " in " << *op);
880  setInPlaceOpResult(result);
881  }
882  }
883  });

886 static LogicalResult bufferizeFuncOpInternals(
887  FuncOp funcOp, BlockAndValueMapping &bvm,
888  const DenseMap<FuncOp, SmallVector<int64_t>> &tiedResultsMap) {
889  OpBuilder b(funcOp->getContext());
890  /// Start by bufferizing `funcOp` arguments.
891  if (failed(bufferize(b, funcOp, bvm)))
892  return failure();
893  WalkResult result = funcOp.walk<WalkOrder::PostOrder>([&](Operation *op) {
894  LogicalResult status =
895  llvm::TypeSwitch<Operation *, LogicalResult>(op)
896  // Skip BufferCast and TensorLoad ops.
897  // clang-format off
898  .Case<memref::BufferCastOp,
899  memref::TensorLoadOp>(
900  [&](auto) { return success(); })
901  .Case<memref::DimOp,
902  LinalgOp,
903  ReturnOp,
904  SubTensorOp,
905  SubTensorInsertOp,
906  VectorTransferOpInterface>(
907  [&](auto op) { return bufferize(b, op, bvm); })
908  // clang-format on
909  .Default([&](Operation *op) {
910  auto isaTensor = [](Type t) { return t.isa<TensorType>(); };
911  if (llvm::any_of(op->getOperandTypes(), isaTensor) ||
912  llvm::any_of(op->getResultTypes(), isaTensor))
913  return failure();
914  return success();
915  });
916  if (failed(status)) {
917  op->emitError("Failed bufferization");
918  return WalkResult::interrupt();
919  }
920  return WalkResult::advance();
921  });
922  if (result.wasInterrupted())
923  return failure();
924  return success();
925 }
