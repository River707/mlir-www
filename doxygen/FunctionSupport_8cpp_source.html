15 /// Helper to call a callback once on each index in the range
16 /// [0, `totalIndices`), *except* for the indices given in `indices`.
17 /// `indices` is allowed to have duplicates and can be in any order.
18 inline void iterateIndicesExcept(unsigned totalIndices,
19  ArrayRef<unsigned> indices,
20  function_ref<void(unsigned)> callback) {
21  llvm::BitVector skipIndices(totalIndices);
22  for (unsigned i : indices)
23  skipIndices.set(i);
24 
25  for (unsigned i = 0; i < totalIndices; ++i)
26  if (!skipIndices.test(i))
27  callback(i);
28 }

54 void mlir::function_like_impl::detail::setArgResAttrDict(
55  Operation *op, StringRef attrName, unsigned numTotalIndices, unsigned index,
56  DictionaryAttr attrs) {
57  ArrayAttr allAttrs = op->getAttrOfType<ArrayAttr>(attrName);
58  if (!allAttrs) {
59  if (attrs.empty())
60  return;
61 
62  // If this attribute is not empty, we need to create a new attribute array.
63  SmallVector<Attribute, 8> newAttrs(numTotalIndices,
64  DictionaryAttr::get(op->getContext()));
65  newAttrs[index] = attrs;
66  op->setAttr(attrName, ArrayAttr::get(op->getContext(), newAttrs));
67  return;
68  }
69  // Check to see if the attribute is different from what we already have.
70  if (allAttrs[index] == attrs)
71  return;
72 
73  // If it is, check to see if the attribute array would now contain only empty
74  // dictionaries.
75  ArrayRef<Attribute> rawAttrArray = allAttrs.getValue();
76  if (attrs.empty() &&
77  llvm::all_of(rawAttrArray.take_front(index), isEmptyAttrDict) &&
78  llvm::all_of(rawAttrArray.drop_front(index + 1), isEmptyAttrDict)) {
79  op->removeAttr(attrName);
80  return;
81  }
82 
83  // Otherwise, create a new attribute array with the updated dictionary.
84  SmallVector<Attribute, 8> newAttrs(rawAttrArray.begin(), rawAttrArray.end());
85  newAttrs[index] = attrs;
86  op->setAttr(attrName, ArrayAttr::get(op->getContext(), newAttrs));
87 }

124 void mlir::function_like_impl::eraseFunctionArguments(
125  Operation *op, ArrayRef<unsigned> argIndices, unsigned originalNumArgs,
126  Type newType) {
127  // There are 3 things that need to be updated:
128  // - Function type.
129  // - Arg attrs.
130  // - Block arguments of entry block.
131  Block &entry = op->getRegion(0).front();
132 
133  // Update the argument attributes of the function.
134  if (auto argAttrs = op->getAttrOfType<ArrayAttr>(getArgDictAttrName())) {
135  SmallVector<DictionaryAttr, 4> newArgAttrs;
136  newArgAttrs.reserve(argAttrs.size());
137  iterateIndicesExcept(originalNumArgs, argIndices, [&](unsigned i) {
138  newArgAttrs.emplace_back(argAttrs[i].cast<DictionaryAttr>());
139  });
140  setAllArgAttrDicts(op, newArgAttrs);
141  }
142 
143  // Update the function type and any entry block arguments.
144  op->setAttr(getTypeAttrName(), TypeAttr::get(newType));
145  entry.eraseArguments(argIndices);
146 }

149  Operation *op, ArrayRef<unsigned> resultIndices,
150  unsigned originalNumResults, Type newType) {
151  // There are 2 things that need to be updated:
152  // - Function type.
153  // - Result attrs.

162  setAllResultAttrDicts(op, newResultAttrs);

164 
165  // Update the function type.
166  op->setAttr(getTypeAttrName(), TypeAttr::get(newType));

180 void mlir::function_like_impl::setFunctionType(Operation *op,
181  FunctionType newType) {
182  assert(op->hasTrait<OpTrait::FunctionLike>());
183  FunctionType oldType = getFunctionType(op);
184  op->setAttr(getTypeAttrName(), TypeAttr::get(newType));
185 
186  // Functor used to update the argument and result attributes of the function.
187  auto updateAttrFn = [&](StringRef attrName, unsigned oldCount,
188  unsigned newCount, auto setAttrFn) {
189  if (oldCount == newCount)
190  return;
191  // The new type has no arguments/results, just drop the attribute.
192  if (newCount == 0) {
193  op->removeAttr(attrName);
194  return;
195  }
196  ArrayAttr attrs = op->getAttrOfType<ArrayAttr>(attrName);
197  if (!attrs)
198  return;
199 
200  // The new type has less arguments/results, take the first N attributes.
201  if (newCount < oldCount)
202  return setAttrFn(op, attrs.getValue().take_front(newCount));
203 
204  // Otherwise, the new type has more arguments/results. Initialize the new
205  // arguments/results with empty attributes.
206  SmallVector<Attribute> newAttrs(attrs.begin(), attrs.end());
207  newAttrs.resize(newCount);
208  setAttrFn(op, newAttrs);
209  };
210 
211  // Update the argument and result attributes.
212  updateAttrFn(function_like_impl::getArgDictAttrName(), oldType.getNumInputs(),
213  newType.getNumInputs(), [&](Operation *op, auto &&attrs) {
214  setAllArgAttrDicts(op, attrs);
215  });
216  updateAttrFn(
217  function_like_impl::getResultDictAttrName(), oldType.getNumResults(),
218  newType.getNumResults(),
219  [&](Operation *op, auto &&attrs) { setAllResultAttrDicts(op, attrs); });
220 }
