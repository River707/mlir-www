34 // The store to load forwarding and load CSE rely on three conditions:
35 //
36 // 1) store/load and load need to have mathematically equivalent affine access
37 // functions (checked after full composition of load/store operands); this
38 // implies that they access the same single memref element for all iterations of
39 // the common surrounding loop,
40 //
41 // 2) the store/load op should dominate the load op,
42 //
43 // 3) among all op's that satisfy both (1) and (2), for store to load
44 // forwarding, the one that postdominates all store op's that have a dependence
45 // into the load, is provably the last writer to the particular memref location
46 // being loaded at the load op, and its store value can be forwarded to the
47 // load; for load CSE, any op that postdominates all store op's that have a
48 // dependence into the load can be forwarded and the first one found is chosen.
49 // Note that the only dependences that are to be considered are those that are
50 // satisfied at the block* of the innermost common surrounding loop of the
51 // <store/load, load> being considered.
52 //
53 // (* A dependence being satisfied at a block: a dependence that is satisfied by
54 // virtue of the destination operation appearing textually / lexically after
55 // the source operation within the body of a 'affine.for' operation; thus, a
56 // dependence is always either satisfied by a loop or by a block).
57 //
58 // The above conditions are simple to check, sufficient, and powerful for most
59 // cases in practice - they are sufficient, but not necessary --- since they
60 // don't reason about loops that are guaranteed to execute at least once or
61 // multiple sources to forward from.
62 //
63 // TODO: more forwarding can be done when support for
64 // loop/conditional live-out SSA values is available.
65 // TODO: do general dead store elimination for memref's. This pass
66 // currently only eliminates the stores only if no other loads/uses (other
67 // than dealloc) remain.
68 //
69 struct MemRefDataFlowOpt : public MemRefDataFlowOptBase<MemRefDataFlowOpt> {

93 // Check if the store may be reaching the load.
94 static bool storeMayReachLoad(Operation *storeOp, Operation *loadOp,
95  unsigned minSurroundingLoops) {
96  MemRefAccess srcAccess(storeOp);
97  MemRefAccess destAccess(loadOp);
98  FlatAffineConstraints dependenceConstraints;
99  unsigned nsLoops = getNumCommonSurroundingLoops(*loadOp, *storeOp);
100  unsigned d;
101  // Dependences at loop depth <= minSurroundingLoops do NOT matter.
102  for (d = nsLoops + 1; d > minSurroundingLoops; d--) {
103  DependenceResult result = checkMemrefAccessDependence(
104  srcAccess, destAccess, d, &dependenceConstraints,
105  /*dependenceComponents=*/nullptr);
106  if (hasDependence(result))
107  break;
108  }
109  if (d <= minSurroundingLoops)
110  return false;
111 
112  return true;
113 }

116 // if needed.

118 MemRefDataFlowOpt::forwardStoreToLoad(AffineReadOpInterface loadOp) {
119  // First pass over the use list to get the minimum number of surrounding
120  // loops common between the load op and the store op, with min taken across
121  // all store ops.
122  SmallVector<Operation *, 8> storeOps;
123  unsigned minSurroundingLoops = getNestingDepth(loadOp);
124  for (auto *user : loadOp.getMemRef().getUsers()) {
125  auto storeOp = dyn_cast<AffineWriteOpInterface>(user);
126  if (!storeOp)
127  continue;
128  unsigned nsLoops = getNumCommonSurroundingLoops(*loadOp, *storeOp);
129  minSurroundingLoops = std::min(nsLoops, minSurroundingLoops);
130  storeOps.push_back(storeOp);
131  }
132 
133  // The list of store op candidates for forwarding that satisfy conditions
134  // (1) and (2) above - they will be filtered later when checking (3).
135  SmallVector<Operation *, 8> fwdingCandidates;
136 
137  // Store ops that have a dependence into the load (even if they aren't
138  // forwarding candidates). Each forwarding candidate will be checked for a
139  // post-dominance on these. 'fwdingCandidates' are a subset of depSrcStores.
140  SmallVector<Operation *, 8> depSrcStores;
141 
142  for (auto *storeOp : storeOps) {
143  if (!storeMayReachLoad(storeOp, loadOp, minSurroundingLoops))
144  continue;
145 
146  // Stores that *may* be reaching the load.
147  depSrcStores.push_back(storeOp);
148 
149  // 1. Check if the store and the load have mathematically equivalent
150  // affine access functions; this implies that they statically refer to the
151  // same single memref element. As an example this filters out cases like:
152  // store %A[%i0 + 1]
153  // load %A[%i0]
154  // store %A[%M]
155  // load %A[%N]
156  // Use the AffineValueMap difference based memref access equality checking.
157  MemRefAccess srcAccess(storeOp);
158  MemRefAccess destAccess(loadOp);
159  if (srcAccess != destAccess)
160  continue;
161 
162  // 2. The store has to dominate the load op to be candidate.
163  if (!domInfo->dominates(storeOp, loadOp))
164  continue;
165 
166  // We now have a candidate for forwarding.
167  fwdingCandidates.push_back(storeOp);
168  }
169 
170  // 3. Of all the store op's that meet the above criteria, the store that
171  // postdominates all 'depSrcStores' (if one exists) is the unique store
172  // providing the value to the load, i.e., provably the last writer to that
173  // memref loc.
174  // Note: this can be implemented in a cleaner way with postdominator tree
175  // traversals. Consider this for the future if needed.
176  Operation *lastWriteStoreOp = nullptr;
177  for (auto *storeOp : fwdingCandidates) {
178  if (llvm::all_of(depSrcStores, [&](Operation *depStore) {
179  return postDomInfo->postDominates(storeOp, depStore);
180  })) {
181  lastWriteStoreOp = storeOp;
182  break;
183  }
184  }
185  if (!lastWriteStoreOp)
186  return failure();
187 
188  // Perform the actual store to load forwarding.
189  Value storeVal =
190  cast<AffineWriteOpInterface>(lastWriteStoreOp).getValueToStore();
191  // Check if 2 values have the same shape. This is needed for affine vector
192  // loads and stores.
193  if (storeVal.getType() != loadOp.getValue().getType())
194  return failure();
195  loadOp.getValue().replaceAllUsesWith(storeVal);
196  // Record the memref for a later sweep to optimize away.
197  memrefsToErase.insert(loadOp.getMemRef());
198  // Record this to erase later.
199  loadOpsToErase.push_back(loadOp);
200  return success();
201 }

203 // The load to load forwarding / redundant load elimination is similar to the
204 // store to load forwarding.
205 // loadA will be be replaced with loadB if:
206 // 1) loadA and loadB have mathematically equivalent affine access functions.
207 // 2) loadB dominates loadA.
208 // 3) loadB postdominates all the store op's that have a dependence into loadA.
209 void MemRefDataFlowOpt::loadCSE(AffineReadOpInterface loadOp) {
210  // The list of load op candidates for forwarding that satisfy conditions
211  // (1) and (2) above - they will be filtered later when checking (3).
212  SmallVector<Operation *, 8> fwdingCandidates;
213  SmallVector<Operation *, 8> storeOps;
214  unsigned minSurroundingLoops = getNestingDepth(loadOp);
215  MemRefAccess memRefAccess(loadOp);
216  // First pass over the use list to get 1) the minimum number of surrounding
217  // loops common between the load op and an load op candidate, with min taken
218  // across all load op candidates; 2) load op candidates; 3) store ops.
219  // We take min across all load op candidates instead of all load ops to make
220  // sure later dependence check is performed at loop depths that do matter.
221  for (auto *user : loadOp.getMemRef().getUsers()) {
222  if (auto storeOp = dyn_cast<AffineWriteOpInterface>(user)) {
223  storeOps.push_back(storeOp);
224  } else if (auto aLoadOp = dyn_cast<AffineReadOpInterface>(user)) {
225  MemRefAccess otherMemRefAccess(aLoadOp);
226  // No need to consider Load ops that have been replaced in previous store
227  // to load forwarding or loadCSE. If loadA or storeA can be forwarded to
228  // loadB, then loadA or storeA can be forwarded to loadC iff loadB can be
229  // forwarded to loadC.
230  // If loadB is visited before loadC and replace with loadA, we do not put
231  // loadB in candidates list, only loadA. If loadC is visited before loadB,
232  // loadC may be replaced with loadB, which will be replaced with loadA
233  // later.
234  if (aLoadOp != loadOp && !llvm::is_contained(loadOpsToErase, aLoadOp) &&
235  memRefAccess == otherMemRefAccess &&
236  domInfo->dominates(aLoadOp, loadOp)) {
237  fwdingCandidates.push_back(aLoadOp);
238  unsigned nsLoops = getNumCommonSurroundingLoops(*loadOp, *aLoadOp);
239  minSurroundingLoops = std::min(nsLoops, minSurroundingLoops);
240  }
241  }
242  }
243 
244  // No forwarding candidate.
245  if (fwdingCandidates.empty())
246  return;
247 
248  // Store ops that have a dependence into the load.
249  SmallVector<Operation *, 8> depSrcStores;
250 
251  for (auto *storeOp : storeOps) {
252  if (!storeMayReachLoad(storeOp, loadOp, minSurroundingLoops))
253  continue;
254 
255  // Stores that *may* be reaching the load.
256  depSrcStores.push_back(storeOp);
257  }
258 
259  // 3. Of all the load op's that meet the above criteria, return the first load
260  // found that postdominates all 'depSrcStores' and has the same shape as the
261  // load to be replaced (if one exists). The shape check is needed for affine
262  // vector loads.
263  Operation *firstLoadOp = nullptr;
264  Value oldVal = loadOp.getValue();
265  for (auto *loadOp : fwdingCandidates) {
266  if (llvm::all_of(depSrcStores,
267  [&](Operation *depStore) {
268  return postDomInfo->postDominates(loadOp, depStore);
269  }) &&
270  cast<AffineReadOpInterface>(loadOp).getValue().getType() ==
271  oldVal.getType()) {
272  firstLoadOp = loadOp;
273  break;
274  }
275  }
276  if (!firstLoadOp)
277  return;
278 
279  // Perform the actual load to load forwarding.
280  Value loadVal = cast<AffineReadOpInterface>(firstLoadOp).getValue();
281  loadOp.getValue().replaceAllUsesWith(loadVal);
282  // Record this to erase later.
283  loadOpsToErase.push_back(loadOp);
284 }

286 void MemRefDataFlowOpt::runOnFunction() {
287  // Only supports single block functions at the moment.
288  FuncOp f = getFunction();
289  if (!llvm::hasSingleElement(f)) {
290  markAllAnalysesPreserved();
291  return;
292  }
293 
294  domInfo = &getAnalysis<DominanceInfo>();
295  postDomInfo = &getAnalysis<PostDominanceInfo>();
296 
297  loadOpsToErase.clear();
298  memrefsToErase.clear();
299 
300  // Walk all load's and perform store to load forwarding and loadCSE.
301  f.walk([&](AffineReadOpInterface loadOp) {
302  // Do store to load forwarding first, if no success, try loadCSE.
303  if (failed(forwardStoreToLoad(loadOp)))
304  loadCSE(loadOp);
305  });
306 
307  // Erase all load op's whose results were replaced with store or load fwd'ed
308  // ones.
309  for (auto *loadOp : loadOpsToErase)
310  loadOp->erase();
311 
312  // Check if the store fwd'ed memrefs are now left with only stores and can
313  // thus be completely deleted. Note: the canonicalize pass should be able
314  // to do this as well, but we'll do it here since we collected these anyway.
315  for (auto memref : memrefsToErase) {
316  // If the memref hasn't been alloc'ed in this function, skip.
317  Operation *defOp = memref.getDefiningOp();
318  if (!defOp || !isa<memref::AllocOp>(defOp))
319  // TODO: if the memref was returned by a 'call' operation, we
320  // could still erase it if the call had no side-effects.
321  continue;
322  if (llvm::any_of(memref.getUsers(), [&](Operation *ownerOp) {
323  return !isa<AffineWriteOpInterface, memref::DeallocOp>(ownerOp);
324  }))
325  continue;
326 
327  // Erase all stores, the dealloc, and the alloc on the memref.
328  for (auto *user : llvm::make_early_inc_range(memref.getUsers()))
329  user->erase();
330  defOp->erase();
331  }
